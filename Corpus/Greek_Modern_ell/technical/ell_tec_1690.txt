def init(self):
 alignment = gtk.Alignment(0.5,0.5,0,0)
 self.click_button = gtk.Button('Click me!')
 alignment.add(self.click_button)
 self.plugin_area.add(alignment)

 self.click_button.connect('clicked', self.onClick)

def accSupportedActions(self):
 try:
 ai = self.node.acc.queryAction()
 except NotImplementedError:
 action_names = []
 else:
 action_names = [ai.getName(i) for i in xrange(ai.nActions)]
 return action_names

def onAccChanged(self, acc):

def onClick(self, button):

Δοκιμάστε την προσιτότητα της εφαρμογής σας
Πρόσθετο προβολής διεπαφής
Το πρόσθετο προβολής διεπαφής εμφανίζει τις πληροφορίες και τους ελέγχους που παρέχονται από τις διεπαφές AT-SPI που βρίσκονται σε κάθε προσιτό αντικείμενο από τη
Εξετάζοντας ένα μενού με το πρόσθετο προβολής διεπαφής
Δίνοντας ένα επιλεγμένο αντικείμενο στη δενδρική προβολή εφαρμογών, όλες οι διεπαφές AT-SPI θα εμφανίζονται ενεργοποιημένες στο πρόσθετο προβολής διεπαφής. Αυτές που δεν υποστηρίζονται θα φανούν ως γκρι, όπως οι διεπαφές της εφαρμογής και συλλογής.
Εκτέλεση του πρόσθετου προβολής διεπαφής
Το πρώτο βήμα για να εκτελέσετε το πρόσθετο προβολής διεπαφής είναι να επιλέξετε ένα αντικείμενο από τη
Accessible - Η διεπαφή αυτή εμφανίζει τον αριθμό των επιλεγμένων θυγατρικών ενός γραφικού συστατικού, την περιγραφή, τις καταστάσεις, τις σχέσεις και τις ιδιότητες.
Ενέργεια - Η διεπαφή αυτή εμφανίζει όλες τις ενέργειες που εκτελούνται από τα επιλεγμένα αντικείμενα. Για κάθε διεπαφή, θα δείτε το όνομα της,όπως "πατήστε" ή "κλικ", ακολουθούμενη, αν υπάρχουν, από μια περιγραφή και έναν συνδυασμό πλήκτρων. Αν επιλέξετε μια ενέργεια, θα γίνει ενεργό ένα κουμπί "Εκτέλεση ενέργειας". Πατήστε αυτό το κουμπί για δείτε το αντικείμενα να εκτελεί την επιλεγμένη ενέργεια.
Εφαρμογή - H διεπαφή αυτή υλοποιείται από αντικείμενα με ρόλο "εφαρμογή". Αξίζει να επισημανθεί ότι αντικείμενα με ρόλο "εφαρμογή" δεν υλοποιούν απαραίτητα τη διεπαφή. Αν το επιλεγμένο αντικείμενο την υλοποιεί, η διεπαφή θα εμφανίσει το αναγνωριστικό της, αν υπάρχει, την εργαλειοθήκη και την έκδοση της.
Συστατικό - Η διεπαφή αυτή εμφανίζει την απόλυτη θέση του επιλεγμένου αντικειμένου (σε σχέση με το σύστημα συντεταγμένων της επιφάνειας εργασίας), τη σχετική του τοποθεσία (σε σχέση με το σύστημα συντεταγμένων των παραθύρων), το μέγεθος του, τον τύπο στης στρώσης (layer), τη σειρά MDI-Z (ένας ακέραιος αριθμός που δείχνει τη σειρά της στοίβας στρώση MDI), και το άλφα (αν χρησιμοποιείται ανάμειξη άλφα).
Έγγραφο - Η διεπαφή αυτή εμφανίζει, αν υπάρχουν, τις ιδιότητες εγγράφου του επιλεγμένου αντικειμένου και την περιγραφή της γλώσσας.
Υπερκείμενο - Η διεπαφή αυτή εμφανίζει μια λίστα με όλους τους υπερσυνδέσμους του επιλεγμένου αντικειμένου. Για κάθε υπερσύνδεσμο, η διεπαφή παρέχει το όνομα του, τη διαδρομή του, την αρχή (το ευρετήριο υπερκειμένου στο οποίο ξεκινάει ο σύνδεσμος) και το τέλος του (το ευρετήριο υπερκειμένου στο οποίο τελειώνει ο σύνδεσμος).
Εικόνα - Η διεπαφή αυτή εμφανίζει της περιγραφή (ή "χωρίς περιγραφή" αν είναι ακαθόριστη) του επιλεγμένου αντικειμένου, το μέγεθος, τη θέση (με βάση το σύστημα συντεταγμένων της επιφάνειας εργασίας) και τη γλώσσα αν υπάρχει μία.
Επιλογή - Η διεπαφή αυτή εμφανίζει τα επιλέξιμα θυγατρικά ενός επιλεγμένου αντικειμένου. Αν το επιλεγμένο αντικείμενο είναι μια γραμμή μενού με επιλογές όπως "Αρχείο", "Επεξεργασία", "Προβολή", και "Βοήθεια", και όλα τα αντικείμενα είναι επιλέξιμα, τότε θα εισαχθούν στη λίστα της διεπαφής. Αν κάνετε κλικ σε μια επιλογή για παράδειγμα - "Αρχείο" - και θέλετε να την καθαρίσετε, πατήστε το κουμπί "Εκκαθάριση" κάτω από τη λίστα.
Περιεχόμενο ροής - Αυτή η διεπαφή εμφανίζει τα περιεχόμενα των επιλεγμένων αντικειμένων και τις αντίστοιχες τους τοποθεσίες.
Πίνακας - Η διεπαφή αυτή εμφανίζει την περιγραφή του επιλεγμένου αντικειμένου, τις γραμμές (αριθμός των γραμμών), τις στήλες (αριθμός των στηλών), τις επιλεγμένες γραμμές (τον αριθμό τους), και τις επιλεγμένες στήλες (τον αριθμό τους). Για κάθε επιλεγμένο κελί, η διεπαφή εμφανίζει τις γραμμές και τις στήλες της κεφαλίδας του.
Κείμενο - Η διεπαφή αυτή εμφανίζει το περιεχόμενο των επιλεγμένων αντικειμένων που μπορεί να επιλεχθεί καθώς και τις ιδιότητες του. Οι βασικές του ιδιότητες είναι offset (ο αριθμός των χαρακτήρων που έχει το κείμενο) και μια αιτιολόγιση. Για να δείτε πιθανές τιμές της CSS, επιλέξτε το κουτί επιλογής "Include defaults"
Τιμή - Η διεπαφή αυτή εμφανίζει την τρέχουσα τιμή του επιλεγμένου αντικειμένου, την ελάχιστη επιτρεπόμενη τιμή, τη μέγιστη επιτρεπόμενη τιμή και την ελάχιστη προσαύξηση στην οποία μπορεί να αλλαχθεί η τιμή (ή μηδέν αν η τιμή δεν μπορεί να καθοριστεί). Η τρέχουσα τιμή μπορεί να αλλαχθεί από το περιστρεφόμενο κουμπί._

3. Πρόσθετο επικύρωσης AT-SPI
Δοκιμή μιας συσκευής για προσβασιμότητα
Πρόσθετο επικύρωσης AT-SPI
Το πρόσθετο επικύρωσης AT-SPI δημιουργεί μια αναφορά για τα προβλήματα προσβασιμότητας μιας εφαρμογής. Για κάθε σφάλμα, η αναφορά περιλαμβάνει το επίπεδο της σοβαρότητας και μια σύντομη περιγραφή.
Το πρόσθετο επικύρωσης AT-SPI δημιουργεί χρήσιμες αναφορές για τα προβλήματα προσβασιμότητας μιας εφαρμογής.
Εκτέλεση του πρόσθετου επικύρωσης AT-SPI
Για να εκτελέσετε το πρόσθετο επικύρωσης AT-SPI, επιλέξτε ένα αντικείμενο από τη δενδρική προβολή εφαρμογών. Ο επικυρωτής θα δημιουργήσει την αναφορά σε σχέση με το επιλεγμένο αντικείμενο και όλα τα γραφικά συστατικά που βρίσκονται στην ιεραρχία του δέντρου.
Αφού επιλέξετε ένα αντικείμενο, επιλέξτε τη μέθοδο επικύρωσης από το κουτί επιλογών και πατήστε το κουμπί
Οι δημιουργημένες αναφορές είναι διαδραστικές. Επομένως, μπορείτε να κάνετε διπλό κλικ σε οποιοδήποτε σφάλμα και να ελέγξετε πιο γραφικό συστατικό το προκαλεί. Θα εμφανιστεί το αντίστοιχο γραφικό συστατικό επιλεγμένο στη
Αν θέλετε να αποθηκεύσετε την δημιουργημένη αναφορά στο σύστημα σας, κάντε κλικ στο κουμπί
Για να πάρετε περισσότερες πληροφορίες για σφάλματα που δημιουργήθηκαν στην αναφορά, σιγουρευτείτε πως είναι επιλεγμένο το αντίστοιχο αντικείμενο και πατήστε το κουμπί
