#!/bin/sh
 (
 echo "10" ; sleep 1
 echo "# Ενημέρωση καταγραφών αλληλογραφίας" ; sleep 1
 echo "20" ; sleep 1
 echo "# Επαναφορά εργασιών cron" ; sleep 1
 echo "50" ; sleep 1
 echo "This line will just be ignored" ; sleep 1
 echo "75" ; sleep 1
 echo "# Επανεκκίνηση συστήματος" ; sleep 1
 echo "100" ; sleep 1
 ) |
 zenity --progress \\
 --title="Ενημέρωση καταγραφών συστήματος" \\
 --text="Ενημέρωση καταγραφών αλληλογραφίας..." \\
 --percentage=0

 if [ "$?" = -1 ] ; then
 zenity --error \\
 --text="Ακύρωση ενημέρωσης."
 fi

Επιτρέπει την επεξεργασία του προβαλλόμενου κειμένου. Το επεξεργασμένο κείμενο αποδίδεται σαν κανονική έξοδος όταν κλείνει ο διάλογος.
#!/bin/sh

 FILE=`zenity --file-selection \\
 --title="Επιλογή αρχείου"`

 case $? in
 0)
 zenity --text-info \\
 --title=$FILE \\
 --filename=$FILE \\
 --editable 2gt;/tmp/tmp.txt;;
 1)
 echo "Δεν επιλέχθηκε αρχείο.";;
 -1)
 echo "Δεν επιλέχθηκε αρχείο.";;
 esac
