Function Hex2Int( sHex As String ) As Long
REM 16 進数値をロング整数に換算して返す
Hex2Int = clng( sHex )
End Function
Function Int2Hex( iLong As Long) As String
REM 整数値を 16 進数値に換算
Int2Hex = &quot; & H &quot; & Hex( iLong )
End Function
